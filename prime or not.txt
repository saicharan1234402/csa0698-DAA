#include<iostream>
#include<cmath>
using namespace std;

int main() {
    int n, i, count = 0;
    cout << "Enter a number: ";
    cin >> n;

    for (i = 1; i <= sqrt(n); i++) {
        if (n % i == 0) {
            count++;
        }
    }

    if (count == 1) { 
        cout << "It is a prime number";
    } else {
        cout << "It is not a prime number";
    }

    return 0;
}
